create table user(
	UUID varchar(40) primary key,
	username varchar(40) not null,
	password varchar(40)not null,
	sex varchar(10),
	phone varchar(20),
	email varchar(40),
	time varchar(20),
	power int

create table resume(
	UUID varchar(40) primary key, 
	userid varchar(40),
	rname varchar(40),
	sex varchar(2),
	picture varchar(100),
	birthday varchar(20),
	nation varchar(20),
	major varchar(100),
	workexperience varchar(100),
	email varchar(40),
	phone varchar(11),
	happy varchar(40),
	education varchar(20),
	advantage varchar(100),
	jobintension varchar(100),
	jobaddr varchar(40),
	honor varchar(100),
	foreign key(userid) references user(UUID)
);

create table company(
	UUID varchar(40) primary key,
	cname varchar(40),
	manager varchar(40),
	position varchar(40),
	phone varchar(13),
	time varchar(11),
	countpeople int
); 
//ְλ
create table post(
	UUID varchar(40) primary key,
	name varchar(40)
);
//cp
create table company_post(
	company_id varchar(40),
	post_id varchar(40),
	price int,
	intro varchar(100),
	primary key(company_id,post_id),
	foreign key(company_id) references company(UUID),
	foreign key(post_id) references post(UUID)
);
//up
create table user_post(
	user_id varchar(40),
	post_id varchar(40),
	primary key(user_id,post_id),
	foreign key(user_id) references user(UUID),
	foreign key(post_id) references post(UUID)
);
